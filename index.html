// ================= FULL JUMIA-LIKE PROJECT ================= // STRUCTURE: // /project // server.js // /models // /routes // /public/index.html // ================= BACKEND (server.js) ================= const express = require('express'); const cors = require('cors'); const mongoose = require('mongoose'); const app = express(); app.use(cors()); app.use(express.json()); app.use(express.static('public')); // MongoDB connect mongoose.connect('mongodb://127.0.0.1:27017/store'); // Product Model const Product = mongoose.model('Product', { name: String, price: Number, img: String }); // User Model const User = mongoose.model('User', { username: String, password: String, role: String }); // ROUTES // GET products app.get('/api/products', async (req, res) => { const products = await Product.find(); res.json(products); }); // ADD product (admin) app.post('/api/products', async (req, res) => { const p = new Product(req.body); await p.save(); res.json(p); }); // DELETE product app.delete('/api/products/:id', async (req, res) => { await Product.findByIdAndDelete(req.params.id); res.json({ success: true }); }); // REGISTER app.post('/api/register', async (req, res) => { const user = new User({ ...req.body, role: 'user' }); await user.save(); res.json(user); }); // LOGIN app.post('/api/login', async (req, res) => { const user = await User.findOne(req.body); if (!user) return res.status(401).json({ error: 'Invalid' }); res.json(user); }); app.listen(3000, () => console.log('Server running')); // ================= FRONTEND (public/index.html) ================= Jumia Clone
